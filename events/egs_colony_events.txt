# transferred any unemployed elites/specialists etc... down to civilians
planet_event = {
	id = egs_colony.10
	hide_window = yes
	is_triggered_only = yes
	immediate = {
        # fix unemployed pops
        every_owned_pop_group = {
            limit = { is_unemployed = yes }
            save_event_target_as = old_pop_group
            prev = {
                create_pop_group = {
                    pop_group = event_target:old_pop_group
                    category = civilian
                    effect = {
                        transfer_pop_amount = {
                            source = event_target:old_pop_group
                            target = this
                            percentage = 1
                        }
                    }
                }
            }
        }
	}
}
