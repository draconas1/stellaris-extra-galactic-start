egs_player1_trigger = {
    OR = {
        has_origin = origin_egs
        has_country_flag = EGS_PLAYER_1
    }
}

egs_player2_trigger = {
    OR = {
        has_origin = origin_egs2
        has_country_flag = EGS_PLAYER_2
    }
}

egs_county_player1 = {
    root = { egs_player1_trigger = yes }
}

egs_county_player2 = {
    root = { egs_player2_trigger = yes }
}

	is_star = no
	has_deposit = no
	is_colonizable = no
	solar_system = {
		egs_select_all_relevant_systems = yes

	}

egs_is_multiplayer = {
	any_country = {
        egs_player2_trigger = yes
	}
}

egs_gigastructures = {
	any_country = {
        egs_player1_trigger = yes
		has_country_flag = EGS_GIGA_YES
	}
}

egs_resources = {
	any_country = {
        egs_player1_trigger = yes
		has_country_flag = EGS_SR_YES
	}
}

egs_link_l_gate = {
	any_country = {
        egs_player1_trigger = yes
		has_country_flag = EGS_LINK_L_GATE
	}
}

egs_select_all_relevant_systems = {
	OR = {
		AND = {
            egs_county_player1 = yes
			has_star_flag = egs_cluster
		}
		AND = {
            egs_county_player2 = yes
			has_star_flag = egs_p2_cluster
		}
	}
}

egs_home_system = {
	OR = {
		has_star_flag = egs_empire_home_system
		has_star_flag = egs_p2_empire_home
	}
}
